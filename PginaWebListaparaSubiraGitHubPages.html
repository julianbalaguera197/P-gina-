       function mostrarVinos(vinos, contenedorId) {
           const contenedor =document.getElementById(contenedorId);
           contenedor.innerHTML = ''; // Limpiar el contenedor antes de agregar vinos

           vinos.forEach(vino => {
               const vinoCard = document.createElement('div');
               vinoCard.className = 'wine-card';
               vinoCard.innerHTML = `
                   <img src="${vino.imagen}" alt="${vino.nombre}" class="card-img-top">
                   <div class="card-body">
                       <h3 class="wine-card-title">${vino.nombre}</h3>
                       <p class="wine-card-description">${vino.descripcion}</p>
                       <a href="#" class="btn btn-primary" data-wine-id="${vino.id}">Ver Detalles</a>
                        <div class="review-container" id="reviews-container-${vino.id}">
                           <h4>Reseñas</h4>
                           ${generarReseñasHTML(vino.reseñas)}
                           <div class="add-review-form">
                               <h5>Agregar Reseña</h5>
                               <form id="review-form-${vino.id}">
                                   <div class="mb-3">
                                       <label for="review-author-${vino.id}" class="form-label">Tu Nombre:</label>
                                       <input type="text" id="review-author-${vino.id}" class="form-control" required>
                                   </div>
                                   <div class="mb-3">
                                       <label for="review-content-${vino.id}" class="form-label">Tu Reseña:</label>
                                       <textarea id="review-content-${vino.id}" class="review-input form-control" rows="3" required></textarea>
                                   </div>
                                   <button type="submit" class="review-submit">Enviar Reseña</button>
                               </form>
                           </div>
                       </div>
                   </div>
               `;
               contenedor.appendChild(vinoCard);

               const detallesButton = vinoCard.querySelector('.btn-primary');
               detallesButton.addEventListener('click', () => mostrarDetallesVino(vino.id));

               const reviewForm = vinoCard.querySelector(`#review-form-${vino.id}`);
               reviewForm.addEventListener('submit', (event) => {
                   event.preventDefault();
                   const authorInput = document.getElementById(`review-author-${vino.id}`);
                   const contentInput = document.getElementById(`review-content-${vino.id}`);
                   const author = authorInput.value;
                   const content = contentInput.value;

                   if (author && content) {
                       const nuevaReseña = {
                           autor: author,
                           fecha: new Date().toLocaleDateString(),
                           comentario: content
                       };

                       vino.reseñas.push(nuevaReseña);
                       guardarVinosEnLocalStorage(); // Guardar los datos actualizados

                       const reseñasContainer = document.getElementById(`reviews-container-${vino.id}`);
                       reseñasContainer.innerHTML = `<h4>Reseñas</h4> ${generarReseñasHTML(vino.reseñas)}`; // Actualizar la visualización de las reseñas

                       // Limpiar el formulario
                       authorInput.value = '';
                       contentInput.value = '';
                       alert('¡Gracias por tu reseña!'); // Mostrar un mensaje de éxito
                   } else {
                       alert('Por favor, completa todos los campos de la reseña.');
                   }
               });
           });
       }

       function generarReseñasHTML(reseñas) {
           let reseñasHTML = '';
           if (reseñas.length === 0) {
               reseñasHTML = '<p>No hay reseñas aún.</p>';
           } else {
               reseñas.forEach(reseña => {
                   reseñasHTML += `
                       <div class="review">
                           <p class="review-author">${reseña.autor}</p>
                           <p class="review-date">${reseña.fecha}</p>
                           <p class="review-content">${reseña.comentario}</p>
                       </div>
                   `;
               });
           }
           return reseñasHTML;
       }

       function mostrarDetallesVino(vinoId) {
           const vino = vinosData.find(v => v.id === vinoId);
           if (vino) {
               alert(`Detalles del vino: ${vino.nombre}\nDescripción: ${vino.descripcion}`); // Esto es un placeholder, aquí iría un modal o una página de detalles
           }
       }

       function filtrarVinos() {
           const searchTerm = document.getElementById('search-input').value.toLowerCase();
           const tipoFilter = document.getElementById('filter-tipo').value.toLowerCase();
           const regionFilter = document.getElementById('filter-region').value.toLowerCase();

           const vinosFiltrados = vinosData.filter(vino => {
               return (
                   vino.nombre.toLowerCase().includes(searchTerm) &&
                   (tipoFilter === '' || vino.tipo === tipoFilter) &&
                   (regionFilter === '' || vino.region === regionFilter)
               );
           });

           mostrarVinosFiltrados(vinosFiltrados);
       }

       function mostrarVinosFiltrados(vinos) {
           const vinosTintos = vinos.filter(vino => vino.tipo === 'tinto');
           const vinosBlancos = vinos.filter(vino => vino.tipo === 'blanco');
           const vinosRosados = vinos.filter(vino => vino.tipo === 'rosado');
           const vinosEspumosos = vinos.filter(vino => vino.tipo === 'espumoso');

           mostrarVinos(vinosTintos, 'vinos-tintos-grid');
           mostrarVinos(vinosBlancos, 'vinos-blancos-grid');
           mostrarVinos(vinosRosados, 'vinos-rosados-grid');
           mostrarVinos(vinosEspumosos, 'vinos-espumosos-grid');
       }

       // Event listeners para la búsqueda y los filtros
       document.getElementById('search-input').addEventListener('input', filtrarVinos);
       document.getElementById('filter-tipo').addEventListener('change', filtrarVinos);
       document.getElementById('filter-region').addEventListener('change', filtrarVinos);

       // Cargar los datos del localStorage al cargar la página
       window.onload = function() {
           const vinosGuardados = localStorage.getItem('vinosData');
           if (vinosGuardados) {
               vinosData = JSON.parse(vinosGuardados);
           }
           mostrarVinosFiltrados(vinosData); // Mostrar todos los vinos al cargar la página
       };

       // Función para guardar los datos en el localStorage
       function guardarVinosEnLocalStorage() {
           localStorage.setItem('vinosData', JSON.stringify(vinosData));
       }
